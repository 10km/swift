51  /**
52   * ThriftCodecManager contains an index of all known ThriftCodec and can create codecs for
53   * unknown types as needed.  Since codec creation can be very expensive only one instance of this
54   * class should be created.
55   */
56  @ThreadSafe
57  public class ThriftCodecManager
58  {
59      private final ThriftCatalog catalog;
60      private final LoadingCache<ThriftType, ThriftCodec<?>> typeCodecs;
61  
62      public ThriftCodecManager(ThriftCodec<?>... codecs)
63      {
64          this(new CompilerThriftCodecFactory(), codecs);
65      }
66  
67      @Inject
68      public ThriftCodecManager(@InternalThriftCodec Set<ThriftCodec<?>> codecs)
69      {
70          this(new CompilerThriftCodecFactory(), codecs);
71      }

74      {
75          this(factory, new ThriftCatalog(), ImmutableSet.copyOf(codecs));
76      }

78      public ThriftCodecManager(ThriftCodecFactory factory, Set<ThriftCodec<?>> codecs)
79      {
80          this(factory, new ThriftCatalog(), codecs);
81      }

83      public ThriftCodecManager(final ThriftCodecFactory factory, final ThriftCatalog catalog, Set<ThriftCodec<?>> codecs)
84      {
85          Preconditions.checkNotNull(factory, "factory is null");
86          Preconditions.checkNotNull(catalog, "catalog is null");
87  
88          this.catalog = catalog;
89  
90          typeCodecs = CacheBuilder.newBuilder().build(new CacheLoader<ThriftType, ThriftCodec<?>>()
91          {
92              public ThriftCodec<?> load(ThriftType type)
93                      throws Exception
94              {
95                  switch (type.getProtocolType()) {
96                      case STRUCT: {
97                          return factory.generateThriftTypeCodec(ThriftCodecManager.this, type.getStructMetadata());
98                      }
99                      case MAP: {
100                         ThriftCodec<?> keyCodec = typeCodecs.get(type.getKeyType());
101                         ThriftCodec<?> valueCodec = typeCodecs.get(type.getValueType());
102                         return new MapThriftCodec<>(type, keyCodec, valueCodec);
103                     }
104                     case SET: {
105                         ThriftCodec<?> elementCodec = typeCodecs.get(type.getValueType());
106                         return new SetThriftCodec<>(type, elementCodec);
107                     }
108                     case LIST: {
109                         ThriftCodec<?> elementCodec = typeCodecs.get(type.getValueType());
110                         return new ListThriftCodec<>(type, elementCodec);
111                     }
112                     case ENUM: {
113                         return new EnumThriftCodec<>(type);
114                     }
115                     default:
116                         if (type.isCoerced()) {
117                             ThriftCodec<?> codec = getCodec(type.getUncoercedType());
118                             TypeCoercion coercion = catalog.getDefaultCoercion(type.getJavaType());
119                             return new CoercionThriftCodec<>(codec, coercion);
120                         }
121                         throw new IllegalArgumentException("Unsupported Thrift type " + type);
122                 }
123             }
124         });
125 
126         addBuiltinCodec(new BooleanThriftCodec());
127         addBuiltinCodec(new ByteThriftCodec());
128         addBuiltinCodec(new ShortThriftCodec());
129         addBuiltinCodec(new IntegerThriftCodec());
130         addBuiltinCodec(new LongThriftCodec());
131         addBuiltinCodec(new DoubleThriftCodec());
132         addBuiltinCodec(new ByteBufferThriftCodec());
133         addBuiltinCodec(new VoidThriftCodec());
134 
135         for (ThriftCodec<?> codec : codecs) {
136             addCodec(codec);
137         }
138     }
139 
140     public ThriftCodec<?> getCodec(Type javaType)
141     {
142         ThriftType thriftType = catalog.getThriftType(javaType);
143         Preconditions.checkArgument(thriftType != null, "Unsupported java type %s", javaType);
144         return getCodec(thriftType);
145     }

147     public <T> ThriftCodec<T> getCodec(Class<T> javaType)
148     {
149         ThriftType thriftType = catalog.getThriftType(javaType);
150         Preconditions.checkArgument(thriftType != null, "Unsupported java type %s", javaType.getName());
151         return (ThriftCodec<T>) getCodec(thriftType);
152     }
153 
154     public ThriftCodec<?> getCodec(ThriftType type)
155     {
156         try {
157             ThriftCodec<?> thriftCodec = typeCodecs.get(type);
158             return thriftCodec;
159         }
160         catch (ExecutionException e) {
161             throw Throwables.propagate(e);
162         }
163     }
164 
165     public <T> ThriftCodec<T> getCodec(TypeToken<T> type)
166     {
167         return (ThriftCodec<T>) getCodec(type.getType());
168     }
169 
170     /**
171      * Adds or replaces the codec associated with the type contained in the codec.  This does not
172      * replace any current users of the existing codec associated with the type.
173      */
174     public void addCodec(ThriftCodec<?> codec)
175     {
176         catalog.addThriftType(codec.getType());
177         typeCodecs.put(codec.getType(), codec);
178     }
179 
180     /**
181      * Adds a ThriftCodec to the codec map, but does not register it with the catalog since builtins
182      * should already be registered
183      */
184     private void addBuiltinCodec(ThriftCodec<?> codec)
185     {
186         typeCodecs.put(codec.getType(), codec);
187     }
188 
189     public ThriftCatalog getCatalog()
190     {
191         return catalog;
192     }

194     public <T> T read(Class<T> type, TProtocol protocol)

196     {
197         return getCodec(type).read(protocol);

200     public Object read(ThriftType type, TProtocol protocol)

202     {

204         return codec.read(protocol);
205     }

207     public <T> void write(Class<T> type, T value, TProtocol protocol)

209     {

211     }

213     public void write(ThriftType type, Object value, TProtocol protocol)

215     {
