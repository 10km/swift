109     @SuppressFBWarnings("DM_DEFAULT_ENCODING")
110     public ThriftCodecByteCodeGenerator(
111             ThriftCodecManager codecManager,
112             ThriftStructMetadata<T> metadata,
113             DynamicClassLoader classLoader,
114             boolean debug
115     )
116     {
117         this.codecManager = codecManager;
118         this.metadata = metadata;
119 
120         structType = type(metadata.getStructClass());
121         codecType = toCodecType(metadata);
122 
123         classDefinition = new ClassDefinition(
124                 a(PUBLIC, SUPER),
125                 codecType.getClassName(),
126                 type(Object.class),
127                 type(ThriftCodec.class, structType)
128         );
129 
130         // declare the class fields
131         typeField = declareTypeField();
132         codecFields = declareCodecFields();
133 
134         // declare methods
135         defineConstructor();
136         defineGetTypeMethod();
137         defineReadMethod();
138         defineWriteMethod();
139 
140         // add the non-generic bridge read and write methods
141         defineReadBridgeMethod();
142         defineWriteBridgeMethod();
143 
144         // generate the byte code
145         ClassWriter cw = new ClassWriter(ClassWriter.COMPUTE_FRAMES);
146         classDefinition.getClassNode().accept(cw);
147         byte[] byteCode = cw.toByteArray();
148 
149         // Run the asm verifier only in debug mode (prints a ton of info)
150         if (debug) {
151             ClassReader reader = new ClassReader(byteCode);
152             CheckClassAdapter.verify(reader, classLoader, true, new PrintWriter(System.out));
153         }
154 
155         // load the class
156         Class<?> codecClass = classLoader.defineClass(codecType.getClassName().replace('/', '.'), byteCode);
157         try {
158             Class<?>[] types = parameters.getTypes();
159             Constructor<?> constructor = codecClass.getConstructor(types);
160             thriftCodec = (ThriftCodec<T>) constructor.newInstance(parameters.getValues());
161         }
162         catch (Exception e) {
163             throw new IllegalStateException("Generated class is invalid", e);
164         }
165     }

167     public ThriftCodec<T> getThriftCodec()
168     {
169         return thriftCodec;
170     }

172     /**
173      * Declares the private ThriftType field type.
174      */
175     private FieldDefinition declareTypeField()
176     {
177         FieldDefinition typeField = new FieldDefinition(a(PRIVATE, FINAL), "type", type(ThriftType.class));
178         classDefinition.addField(typeField);
179 
180         // add constructor parameter to initialize this field
181         parameters.add(typeField, ThriftType.struct(metadata));
182 
183         return typeField;
184     }

186     /**
187      * Declares a field for each delegate codec
188      *
189      * @return a map from field id to the codec for the field
190      */
191     private Map<Short, FieldDefinition> declareCodecFields()
192     {
193         Map<Short, FieldDefinition> codecFields = new TreeMap<>();
194         for (ThriftFieldMetadata fieldMetadata : metadata.getFields()) {
195             if (needsCodec(fieldMetadata)) {
196 
197                 ThriftCodec<?> codec = codecManager.getCodec(fieldMetadata.getType());
198                 String fieldName = fieldMetadata.getName() + "Codec";
199 
200                 FieldDefinition codecField = new FieldDefinition(a(PRIVATE, FINAL), fieldName, type(codec.getClass()));
201                 classDefinition.addField(codecField);
202                 codecFields.put(fieldMetadata.getId(), codecField);
203 
204                 parameters.add(codecField, codec);
205             }
206         }
207         return codecFields;
208     }

210     /**
211      * Defines the constructor with a parameter for the ThriftType and the delegate codecs.  The
212      * constructor simply assigns these parameters to the class fields.
213      */
214     private void defineConstructor()
215     {
216         //
217         // declare the constructor
218         MethodDefinition constructor = new MethodDefinition(
219                 a(PUBLIC),
220                 "<init>",
221                 type(void.class),
222                 parameters.getParameters()
223         );
224 
225         // invoke super (Object) constructor
226         constructor.loadThis().invokeConstructor(type(Object.class));
227 
228         // this.foo = foo;
229         for (FieldDefinition fieldDefinition : parameters.getFields()) {
230             constructor.loadThis()
231                     .loadVariable(fieldDefinition.getName())
232                     .putField(codecType, fieldDefinition);
233         }
234 
235         // return; (implicit)
236         constructor.ret();
237 
238         classDefinition.addMethod(constructor);
239     }

254     /**
255      * Defines the read method.
256      */
257     private void defineReadMethod()
258     {
259         MethodDefinition read = new MethodDefinition(
260                 a(PUBLIC),
261                 "read",
262                 structType,
263                 arg("protocol", TProtocol.class)
264         ).addException(Exception.class);
265 
266         // TProtocolReader reader = new TProtocolReader(protocol);
267         read.addLocalVariable(type(TProtocolReader.class), "reader");
268         read.newObject(TProtocolReader.class);
269         read.dup();
270         read.loadVariable("protocol");
271         read.invokeConstructor(type(TProtocolReader.class), type(TProtocol.class));
272         read.storeVariable("reader");
273 
274         // read all of the data in to local variables
275         Map<Short, LocalVariableDefinition> structData = readFieldValues(read);
276 
277         // build the struct
278         buildStruct(read, structData);
279     }

281     /**
282      * Defines the code to read all of the data from the protocol into local variables.
283      */
284     private Map<Short, LocalVariableDefinition> readFieldValues(MethodDefinition read)
285     {
286         LocalVariableDefinition protocol = read.getLocalVariable("reader");
287 
288         // declare and init local variables here
289         Map<Short, LocalVariableDefinition> structData = new TreeMap<>();
290         for (ThriftFieldMetadata field : metadata.getFields()) {
291             LocalVariableDefinition variable = read.addInitializedLocalVariable(
292                     toParameterizedType(field.getType()),
293                     "f_" + field.getName()
294             );
295             structData.put(field.getId(), variable);
296         }
297 
298         // protocol.readStructBegin();
299         read.loadVariable(protocol).invokeVirtual(
300                 TProtocolReader.class,
301                 "readStructBegin",
302                 void.class
303         );
304 
305         // while (protocol.nextField())
306         read.visitLabel("while-begin");
307         read.loadVariable(protocol).invokeVirtual(TProtocolReader.class, "nextField", boolean.class);
308         read.ifZeroGoto("while-end");
309 
310         // switch (protocol.getFieldId())
311         read.loadVariable(protocol).invokeVirtual(TProtocolReader.class, "getFieldId", short.class);
312         List<CaseStatement> cases = new ArrayList<>();
313         for (ThriftFieldMetadata field : metadata.getFields()) {
314             cases.add(caseStatement(field.getId(), field.getName() + "-field"));
315         }
316         read.switchStatement("default", cases);
317 
318         for (ThriftFieldMetadata field : metadata.getFields()) {
319             // case field.id:
320             read.visitLabel(field.getName() + "-field");
321 
322             // push protocol
323             read.loadVariable(protocol);
324 
325             // push ThriftTypeCodec for this field
326             FieldDefinition codecField = codecFields.get(field.getId());
327             if (codecField != null) {
328                 read.loadThis().getField(codecType, codecField);
329             }
330 
331             // read value
332             Method readMethod = READ_METHODS.get(field.getType().getProtocolType());
333             if (readMethod == null) {
334                 throw new IllegalArgumentException("Unsupported field type " + field.getType().getProtocolType());
335             }
336             read.invokeVirtual(readMethod);
337 
338             // todo this cast should be based on readMethod return type and fieldType (or coercion type)
339             // add cast if necessary
340             if (needsCastAfterRead(field, readMethod)) {
341                 read.checkCast(toParameterizedType(field.getType()));
342             }
343 
344             // coerce the type
345             if (field.getCoercion() != null) {
346                 read.invokeStatic(field.getCoercion().getFromThrift());
347             }
348 
349             // store protocol value
350             read.storeVariable(structData.get(field.getId()));
351 
352             // go back to top of loop
353             read.gotoLabel("while-begin");
354         }
355 
356         // default case
357         read.visitLabel("default")
358                 .loadVariable(protocol)
359                 .invokeVirtual(TProtocolReader.class, "skipFieldData", void.class)
360                 .gotoLabel("while-begin");
361 
362         // end of while loop
363         read.visitLabel("while-end");
364 
365         // protocol.readStructEnd();
366         read.loadVariable(protocol)
367                 .invokeVirtual(TProtocolReader.class, "readStructEnd", void.class);
368         return structData;
369     }

371     /**
372      * Defines the code to build the struct instance using the data in the local variables.
373      */
374     private void buildStruct(MethodDefinition read, Map<Short, LocalVariableDefinition> structData)
375     {
376 
377         // construct the instance and store it in the instance local variable
378         LocalVariableDefinition instance = constructInstance(read, structData);
379 
380         // inject fields
381         injectFields(read, instance, structData);
382 
383         // inject methods
384         injectMethods(read, instance, structData);
385 
386         // invoke factory method if present
387         invokeFactoryMethod(read, structData, instance);
388 
389         // push instance on stack, and return it
390         read.loadVariable(instance)
391                 .retObject();
392 
393         classDefinition.addMethod(read);
394     }

396     /**
397      * Defines the code to construct the struct (or builder) instance and stores it in a local
398      * variable.
399      */
400     private LocalVariableDefinition constructInstance(MethodDefinition read, Map<Short, LocalVariableDefinition> structData)
401     {
402         LocalVariableDefinition instance = read.addLocalVariable(structType, "instance");
403 
404         // create the new instance (or builder)
405         if (metadata.getBuilderClass() == null) {
406             read.newObject(structType).dup();
407         }
408         else {
409             read.newObject(metadata.getBuilderClass()).dup();
410         }
411 
412         // invoke constructor
413         ThriftConstructorInjection constructor = metadata.getConstructor();
414         // push parameters on stack
415         for (ThriftParameterInjection parameter : constructor.getParameters()) {
416             read.loadVariable(structData.get(parameter.getId()));
417         }
418         // invoke constructor
419         read.invokeConstructor(constructor.getConstructor())
420                 .storeVariable(instance);
421         return instance;
422     }

424     /**
425      * Defines the code to inject data into the struct public fields.
426      */
427     private void injectFields(MethodDefinition read, LocalVariableDefinition instance, Map<Short, LocalVariableDefinition> structData)
428     {
429         for (ThriftFieldMetadata field : metadata.getFields()) {
430             for (ThriftInjection injection : field.getInjections()) {
431                 if (injection instanceof ThriftFieldInjection) {
432 
433                     ThriftFieldInjection fieldInjection = (ThriftFieldInjection) injection;
434 
435                     // if field is an Object && field != null
436                     if (!isProtocolTypeJavaPrimitive(field)) {
437                         read.loadVariable(structData.get(field.getId()))
438                                 .ifNullGoto("field_is_null_" + field.getName());
439                     }
440 
441                     // write value
442                     read.loadVariable(instance)
443                             .loadVariable(structData.get(field.getId()))
444                             .putField(fieldInjection.getField());
445 
446                     // else do nothing
447                     if (!isProtocolTypeJavaPrimitive(field)) {
448                         read.visitLabel("field_is_null_" + field.getName());
449                     }
450                 }
451             }
452         }
453     }

455     /**
456      * Defines the code to inject data into the struct methods.
457      */
458     private void injectMethods(MethodDefinition read, LocalVariableDefinition instance, Map<Short, LocalVariableDefinition> structData)
459     {
460         for (ThriftMethodInjection methodInjection : metadata.getMethodInjections()) {
461             // if any parameter is non-null, invoke the method
462             for (ThriftParameterInjection parameter : methodInjection.getParameters()) {
463                 if (!isParameterTypeJavaPrimitive(parameter)) {
464                     read.loadVariable(structData.get(parameter.getId()));
465                     read.ifNotNullGoto("invoke_" + methodInjection.getMethod().toGenericString());
466                 }
467                 else {
468                     read.gotoLabel("invoke_" + methodInjection.getMethod().toGenericString());
469                 }
470             }
471             read.gotoLabel("skip_invoke_" + methodInjection.getMethod().toGenericString());
472 
473             // invoke the method
474             read.visitLabel("invoke_" + methodInjection.getMethod().toGenericString());
475             read.loadVariable(instance);
476 
477             // push parameters on stack
478             for (ThriftParameterInjection parameter : methodInjection.getParameters()) {
479                 read.loadVariable(structData.get(parameter.getId()));
480             }
481 
482             // invoke the method
483             read.invokeVirtual(methodInjection.getMethod());
484 
485             // if method has a return, we need to pop it off the stack
486             if (methodInjection.getMethod().getReturnType() != void.class) {
487                 read.pop();
488             }
489 
490             // skip invocation
491             read.visitLabel("skip_invoke_" + methodInjection.getMethod().toGenericString());
492         }
493     }

495     /**
496      * Defines the code that calls the builder factory method.
497      */
498     private void invokeFactoryMethod(MethodDefinition read, Map<Short, LocalVariableDefinition> structData, LocalVariableDefinition instance)
499     {
500         ThriftMethodInjection builderMethod = metadata.getBuilderMethod();
501         if (builderMethod != null) {
502             read.loadVariable(instance);
503 
504             // push parameters on stack
505             for (ThriftParameterInjection parameter : builderMethod.getParameters()) {
506                 read.loadVariable(structData.get(parameter.getId()));
507             }
508 
509             // invoke the method
510             read.invokeVirtual(builderMethod.getMethod())
511                     .storeVariable(instance);
512         }
513     }

515     /**
516      * Define the write method.
517      */
518     private void defineWriteMethod()
519     {
520         MethodDefinition write = new MethodDefinition(
521                 a(PUBLIC),
522                 "write",
523                 null,
524                 arg("struct", structType),
525                 arg("protocol", TProtocol.class)
526         );
527         classDefinition.addMethod(write);
528 
529         // TProtocolReader reader = new TProtocolReader(protocol);
530         write.addLocalVariable(type(TProtocolWriter.class), "writer");
531         write.newObject(TProtocolWriter.class);
532         write.dup();
533         write.loadVariable("protocol");
534         write.invokeConstructor(type(TProtocolWriter.class), type(TProtocol.class));
535         write.storeVariable("writer");
536 
537 
538         LocalVariableDefinition protocol = write.getLocalVariable("writer");
539 
540         // protocol.writeStructBegin("bonk");
541         write.loadVariable(protocol)
542                 .loadConstant(metadata.getStructName())
543                 .invokeVirtual(TProtocolWriter.class, "writeStructBegin", void.class, String.class);
544 
545         // write fields
546         for (ThriftFieldMetadata field : metadata.getFields()) {
547             // push protocol
548             write.loadVariable(protocol);
549 
550             // push (String) field.name
551             write.loadConstant(field.getName());
552 
553             // push (short) field.id
554             write.loadConstant(field.getId());
555 
556             // push ThriftTypeCodec for this field
557             FieldDefinition codecField = codecFields.get(field.getId());
558             if (codecField != null) {
559                 write.loadThis().getField(codecType, codecField);
560             }
561 
562             // push field value
563             loadFieldValue(write, field);
564 
565             // if field value is null, don't coerce or write the field
566             if (!isFieldTypeJavaPrimitive(field)) {
567                 // ifNullGoto consumes the top of the stack, so we need to duplicate the value
568                 write.dup();
569                 write.ifNullGoto("field_is_null_" + field.getName());
570             }
571 
572             // coerce value
573             if (field.getCoercion() != null) {
574                 write.invokeStatic(field.getCoercion().getToThrift());
575 
576                 // if coerced value is null, don't write the field
577                 if (!isProtocolTypeJavaPrimitive(field)) {
578                     write.dup();
579                     write.ifNullGoto("field_is_null_" + field.getName());
580                 }
581             }
582 
583             // write value
584             Method writeMethod = WRITE_METHODS.get(field.getType().getProtocolType());
585             if (writeMethod == null) {
586                 throw new IllegalArgumentException(
587                         "Unsupported field type " + field.getType().getProtocolType()
588                 );
589             }
590             write.invokeVirtual(writeMethod);
591 
592             //
593             // If not written because of a null, clean-up the stack
594             if (!isProtocolTypeJavaPrimitive(field) || !isFieldTypeJavaPrimitive(field)) {
595 
596                 // value was written so skip cleanup
597                 write.gotoLabel("field_end_" + field.getName());
598 
599                 // cleanup stack for null field value
600                 write.visitLabel("field_is_null_" + field.getName());
601                 // pop value
602                 write.pop();
603                 // pop codec
604                 if (codecField != null) {
605                     write.pop();
606                 }
607                 // pop id
608                 write.pop();
609                 // pop name
610                 write.pop();
611                 // pop protocol
612                 write.pop();
613 
614                 write.visitLabel("field_end_" + field.getName());
615             }
616         }
617 
618         write.loadVariable(protocol)
619                 .invokeVirtual(TProtocolWriter.class, "writeStructEnd", void.class);
620 
621         write.ret();
622     }

624     private void loadFieldValue(MethodDefinition write, ThriftFieldMetadata field)
625     {
626         write.loadVariable("struct");
627         ThriftExtraction extraction = field.getExtraction();
628         if (extraction instanceof ThriftFieldExtractor) {
629             ThriftFieldExtractor fieldExtractor = (ThriftFieldExtractor) extraction;
630             write.getField(fieldExtractor.getField());
631         }
632         else if (extraction instanceof ThriftMethodExtractor) {
633             ThriftMethodExtractor methodExtractor = (ThriftMethodExtractor) extraction;
634             write.invokeVirtual(methodExtractor.getMethod());
635         }
636     }

675     private boolean isParameterTypeJavaPrimitive(ThriftParameterInjection parameter)
676     {
677         return isJavaPrimitive(TypeToken.of(parameter.getJavaType()));
678     }

680     private boolean isFieldTypeJavaPrimitive(ThriftFieldMetadata field)
681     {
682         return isJavaPrimitive(TypeToken.of(field.getType().getJavaType()));
683     }
684 
685     private boolean isProtocolTypeJavaPrimitive(ThriftFieldMetadata field)
686     {
687         if (field.getType().isCoerced()) {
688             return isJavaPrimitive(TypeToken.of(field.getType().getUncoercedType().getJavaType()));
689         }
690         else {
691             return isJavaPrimitive(TypeToken.of(field.getType().getJavaType()));
692         }
693     }

695     private boolean isJavaPrimitive(TypeToken<?> typeToken)
696     {
697         return typeToken
698                 .getRawType()
699                 .isPrimitive();
700     }

702     private static boolean needsCastAfterRead(ThriftFieldMetadata field, Method readMethod)
703     {
704         Class<?> methodReturn = readMethod.getReturnType();
705         Class<?> fieldType;
706         if (field.getCoercion() != null) {
