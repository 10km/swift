76      public String[] lookupParameterNames(AccessibleObject methodOrConstructor) {
77          return lookupParameterNames(methodOrConstructor, true);

80      public String[] lookupParameterNames(AccessibleObject methodOrCtor, boolean throwExceptionIfMissing) {
81  
82          Class<?>[] types = null;
83          Class<?> declaringClass = null;
84          String name = null;
85          if (methodOrCtor instanceof Method) {
86              Method method = (Method) methodOrCtor;
87              types = method.getParameterTypes();
88              name = method.getName();
89              declaringClass = method.getDeclaringClass();
90          } else if (methodOrCtor instanceof Constructor<?>) {
91              Constructor<?> constructor = (Constructor<?>) methodOrCtor;
92              types = constructor.getParameterTypes();
93              declaringClass = constructor.getDeclaringClass();
94              name = "<init>";
95          }
96          else {
97              throw new IllegalStateException(String.format("%s is neither method nor c'tor!", methodOrCtor.getClass().getSimpleName()));
98          }
99  
100         if (types.length == 0) {
101             return EMPTY_NAMES;
102         }
103         InputStream byteCodeStream = getClassAsStream(declaringClass);
104         if (byteCodeStream == null) {
105             if (throwExceptionIfMissing) {
106                 throw new ParameterNamesNotFoundException("Unable to get class bytes");
107             } else {
108                 return Paranamer.EMPTY_NAMES;
109             }
110         }
111         try {
112             ClassReader reader = new ClassReader(byteCodeStream);
113             TypeCollector visitor = new TypeCollector(name, types, throwExceptionIfMissing);
114             reader.accept(visitor);
115             String[] parameterNamesForMethod = visitor.getParameterNamesForMethod();
116             Closeables.closeQuietly(byteCodeStream);
117             return parameterNamesForMethod;
118         } catch (IOException e) {
119             if (throwExceptionIfMissing) {
120                 throw new ParameterNamesNotFoundException("IoException while reading class bytes", e);
121             } else {
122                 return Paranamer.EMPTY_NAMES;
123             }
124         }
125     }

127     private InputStream getClassAsStream(Class<?> clazz) {
128         ClassLoader classLoader = clazz.getClassLoader();
129         if (classLoader == null) {
130             classLoader = ClassLoader.getSystemClassLoader();
131         }
132         return getClassAsStream(classLoader, clazz.getName());
133     }
134 
135     private InputStream getClassAsStream(ClassLoader classLoader, String className) {
136         String name = className.replace('.', '/') + ".class";
137         // better pre-cache all methods otherwise this content will be loaded
138         // multiple times
139         InputStream asStream = classLoader.getResourceAsStream(name);
140         if (asStream == null) {
141             asStream = BytecodeReadingParanamer.class.getResourceAsStream(name);
142         }
143         return asStream;
144     }

225         private String correctTypeName(Type[] argumentTypes, int i) {
226             String s = argumentTypes[i].getClassName();
227             // array notation needs cleanup.
228             StringBuilder braces = new StringBuilder();
229             while (s.endsWith("[]")) {
230                 braces.append("[");
231                 s = s.substring(0, s.length() - 2);
232             }
233             if (braces.length() != 0) {
234                 if (primitives.containsKey(s)) {
235                     s = braces.toString() + primitives.get(s);
236                 } else {
237                 s = braces + "L" + s + ";";
238                 }
239             }
240             return s;
241         }

243         private String[] getParameterNamesForMethod() {
244             if (collector == null) {
245                 return Paranamer.EMPTY_NAMES;
246             }
247             if (!collector.isDebugInfoPresent()) {
248                 if (throwExceptionIfMissing) {
249                     throw new ParameterNamesNotFoundException("Parameter names not found for " + methodName);
250                 } else {
251                     return Paranamer.EMPTY_NAMES;
252                 }
253             }
254             return collector.getResult().split(COMMA);
255         }

260      * Objects of this class collects information from a specific method.
261      *
262      * @author Guilherme Silveira
263      */
264     private static class MethodCollector {
265 
266         private final int paramCount;
267 
268         private final int ignoreCount;
269 
270         private int currentParameter;
271 
272         private final StringBuffer result;
273 
274         private boolean debugInfoPresent;
275 
276         private MethodCollector(int ignoreCount, int paramCount) {
277             this.ignoreCount = ignoreCount;
278             this.paramCount = paramCount;
279             this.result = new StringBuffer();
280             this.currentParameter = 0;
281             // if there are 0 parameters, there is no need for debug info
282             this.debugInfoPresent = paramCount == 0;
283         }
284 
285         public void visitLocalVariable(String name, int index) {
286             if (index >= ignoreCount && index < ignoreCount + paramCount) {
287                 if (!name.equals("arg" + currentParameter)) {
288                     debugInfoPresent = true;
289                 }
290                 result.append(',');
291                 result.append(name);
292                 currentParameter++;
293             }
294         }
295 
296         private String getResult() {
297             return result.length() != 0 ? result.substring(1) : "";
298         }
299 
300         private boolean isDebugInfoPresent() {
301             return debugInfoPresent;
302         }

338      * A Java class parser to make a Class Visitor visit an existing class.
339      * This class parses a byte array conforming to the Java class file format and
340      * calls the appropriate visit methods of a given class visitor for each field,
341      * method and bytecode instruction encountered.
342      *
343      * @author Eric Bruneton
344      * @author Eugene Kuleshov
345      */
346     private static class ClassReader {
347 
348         /**
349          * The class to be parsed. <i>The content of this array must not be
350          * modified. This field is intended for Attribute sub classes, and
351          * is normally not needed by class generators or adapters.</i>
352          */
353         public final byte[] b;
354 
355         /**
356          * The start index of each constant pool item in {@link #b b}, plus one.
357          * The one byte offset skips the constant pool item tag that indicates its
358          * type.
359          */
360         private final int[] items;
361 
362         /**
363          * The String objects corresponding to the CONSTANT_Utf8 items. This cache
364          * avoids multiple parsing of a given CONSTANT_Utf8 constant pool item,
365          * which GREATLY improves performances (by a factor 2 to 3). This caching
366          * strategy could be extended to all constant pool items, but its benefit
367          * would not be so great for these items (because they are much less
368          * expensive to parse than CONSTANT_Utf8 items).
369          */
370         private final String[] strings;
371 
372         /**
373          * Maximum length of the strings contained in the constant pool of the
374          * class.
375          */
376         private final int maxStringLength;
377 
378         /**
379          * Start index of the class header information (access, name...) in
380          * {@link #b b}.
381          */
382         public final int header;
383 
384 
385         /**
386          * The type of CONSTANT_Fieldref constant pool items.
387          */
388         final static int FIELD = 9;
389 
390         /**
391          * The type of CONSTANT_Methodref constant pool items.
392          */
393         final static int METH = 10;

395         /**
396          * The type of CONSTANT_InterfaceMethodref constant pool items.
397          */
398         final static int IMETH = 11;

403         final static int INT = 3;

407          */

409 
410         /**
411          * The type of CONSTANT_Long constant pool items.
412          */
413         final static int LONG = 5;
414 
415         /**
416          * The type of CONSTANT_Double constant pool items.
417          */
418         final static int DOUBLE = 6;

421          * The type of CONSTANT_NameAndType constant pool items.
422          */
423         final static int NAME_TYPE = 12;

435          * Constructs a new {@link ClassReader} object.

443         /**
444          * Constructs a new {@link ClassReader} object.
445          *
446          * @param b   the bytecode of the class to be read.
447          * @param off the start offset of the class data.
448          */
449         private ClassReader(final byte[] b, final int off) {
450             this.b = b;
451             // parses the constant pool
452             items = new int[readUnsignedShort(off + 8)];
453             int n = items.length;
454             strings = new String[n];
455             int max = 0;
456             int index = off + 10;
457             for (int i = 1; i < n; ++i) {
458                 items[i] = index + 1;
459                 int size;
460                 switch (b[index]) {
461                     case FIELD:
462                     case METH:
463                     case IMETH:
464                     case INT:
465                     case FLOAT:
466                     case NAME_TYPE:
467                         size = 5;
468                         break;
469                     case LONG:
470                     case DOUBLE:
471                         size = 9;
472                         ++i;
473                         break;
474                     case UTF8:
475                         size = 3 + readUnsignedShort(index + 1);
476                         if (size > max) {
477                             max = size;
478                         }
479                         break;
480                         // case HamConstants.CLASS:
481                         // case HamConstants.STR:
482                     default:
483                         size = 3;
484                         break;
485                 }
486                 index += size;
487             }
488             maxStringLength = max;
489             // the class header information starts just after the constant pool
490             header = index;
491         }

504         /**
505          * Reads the bytecode of a class.
506          *
507          * @param is an input stream from which to read the class.
508          * @return the bytecode read from the given input stream.
509          * @throws IOException if a problem occurs during reading.
510          */
511         private static byte[] readClass(final InputStream is) throws IOException {
512             if (is == null) {
513                 throw new IOException("Class not found");
514             }
515             try {
516                   byte[] b = new byte[is.available()];
517                   int len = 0;
518                   while (true) {
519                       int n = is.read(b, len, b.length - len);
520                       if (n == -1) {
521                           if (len < b.length) {
522                               byte[] c = new byte[len];
523                               System.arraycopy(b, 0, c, 0, len);
524                               b = c;
525                           }
526                           return b;
527                       }
528                       len += n;
529                       if (len == b.length) {
530                           int last = is.read();
531                           if (last < 0) {
532                               return b;
533                           }
534                           byte[] c = new byte[b.length + 1000];
535                           System.arraycopy(b, 0, c, 0, len);
536                           c[len++] = (byte) last;
537                           b = c;
538                       }
539                   }
540             } finally {
541                 Closeables.closeQuietly(is);
542             }
543         }

549         /**
550          * Makes the given visitor visit the Java class of this {@link ClassReader}.
551          * This class is the one specified in the constructor (see
552          * {@link #ClassReader(byte[]) ClassReader}).
553          *
554          * @param classVisitor the visitor that must visit this class.
555          */
556         private void accept(final TypeCollector classVisitor) {
557             char[] c = new char[maxStringLength]; // buffer used to read strings
558             int i, j, k; // loop variables
559             int u, v, w; // indexes in b
560 
561             String attrName;
562             int anns = 0;
563             int ianns = 0;
564 
565             // visits the header
566             u = header;
567             v = items[readUnsignedShort(u + 4)];
568             int len = readUnsignedShort(u + 6);
569             w = 0;
570             u += 8;
571             for (i = 0; i < len; ++i) {
572                 u += 2;
573             }
574             v = u;
575             i = readUnsignedShort(v);
576             v += 2;
577             for (; i > 0; --i) {
578                 j = readUnsignedShort(v + 6);
579                 v += 8;
580                 for (; j > 0; --j) {
581                     v += 6 + readInt(v + 2);
582                 }
583             }
584             i = readUnsignedShort(v);
585             v += 2;
586             for (; i > 0; --i) {
587                 j = readUnsignedShort(v + 6);
588                 v += 8;
589                 for (; j > 0; --j) {
590                     v += 6 + readInt(v + 2);
591                 }
592             }
593 
594             i = readUnsignedShort(v);
595             v += 2;
596             for (; i > 0; --i) {
597                 v += 6 + readInt(v + 2);
598             }
599 
600             //annotations not needed.
601 
602             // visits the fields
603             i = readUnsignedShort(u);
604             u += 2;
605             for (; i > 0; --i) {
606                 j = readUnsignedShort(u + 6);
607                 u += 8;
608                 for (; j > 0; --j) {
609                     u += 6 + readInt(u + 2);
610                 }
611             }
612 
613             // visits the methods
614             i = readUnsignedShort(u);
615             u += 2;
616             for (; i > 0; --i) {
617                 // inlined in original ASM source, now a method call
618                 u = readMethod(classVisitor, c, u);
619             }
620 
621         }

623         private int readMethod(TypeCollector classVisitor, char[] c, int u) {
624             int v;
625             int w;
626             int j;
627             String attrName;
628             int k;
629             int access = readUnsignedShort(u);
630             String name = readUTF8(u + 2, c);
631             String desc = readUTF8(u + 4, c);
632             v = 0;
633             w = 0;
634 
635             // looks for Code and Exceptions attributes
636             j = readUnsignedShort(u + 6);
637             u += 8;
638             for (; j > 0; --j) {
639                 attrName = readUTF8(u, c);
640                 int attrSize = readInt(u + 2);
641                 u += 6;
642                 // tests are sorted in decreasing frequency order
643                 // (based on frequencies observed on typical classes)
644                 if (attrName.equals("Code")) {
645                     v = u;
646                 }
647                 u += attrSize;
648             }
649             // reads declared exceptions
650             if (w != 0) {
651                 w += 2;
652                 for (j = 0; j < readUnsignedShort(w); ++j) {
653                     w += 2;
654                 }
655             }
656 
657             // visits the method's code, if any
658             MethodCollector mv = classVisitor.visitMethod(access, name, desc);
659 
660             if (mv != null && v != 0) {
661                 int codeLength = readInt(v + 4);
662                 v += 8;
663 
664                 int codeStart = v;
665                 int codeEnd = v + codeLength;
666                 v = codeEnd;
667 
668                 j = readUnsignedShort(v);
669                 v += 2;
670                 for (; j > 0; --j) {
671                     v += 8;
672                 }
673                 // parses the local variable, line number tables, and code
674                 // attributes
675                 int varTable = 0;
676                 int varTypeTable = 0;
677                 j = readUnsignedShort(v);
678                 v += 2;
679                 for (; j > 0; --j) {
680                     attrName = readUTF8(v, c);
681                     if (attrName.equals("LocalVariableTable")) {
682                         varTable = v + 6;
683                     } else if (attrName.equals("LocalVariableTypeTable")) {
684                         varTypeTable = v + 6;
685                     }
686                     v += 6 + readInt(v + 2);
687                 }
688 
689                 v = codeStart;
690                 // visits the local variable tables
691                 if (varTable != 0) {
692                     if (varTypeTable != 0) {
693                         k = readUnsignedShort(varTypeTable) * 3;
694                         w = varTypeTable + 2;
695                         int[] typeTable = new int[k];
696                         while (k > 0) {
697                             typeTable[--k] = w + 6; // signature
698                             typeTable[--k] = readUnsignedShort(w + 8); // index
699                             typeTable[--k] = readUnsignedShort(w); // start
700                             w += 10;
701                         }
702                     }
703                     k = readUnsignedShort(varTable);
704                     w = varTable + 2;
705                     for (; k > 0; --k) {
706                         int index = readUnsignedShort(w + 8);
707                         mv.visitLocalVariable(readUTF8(w + 4, c), index);
708                         w += 10;
709                     }
710                 }
711             }
712             return u;
713         }

715         /**
716          * Reads an unsigned short value in {@link #b b}. <i>This method is
717          * intended for Attribute sub classes, and is normally not needed by
718          * class generators or adapters.</i>
719          *
720          * @param index the start index of the value to be read in {@link #b b}.
721          * @return the read value.
722          */
723         private int readUnsignedShort(final int index) {
724             byte[] b = this.b;
725             return ((b[index] & 0xFF) << 8) | (b[index + 1] & 0xFF);
726         }
727 
728         /**
729          * Reads a signed int value in {@link #b b}. <i>This method is intended for
730          * Attribute sub classes, and is normally not needed by class
731          * generators or adapters.</i>
732          *
733          * @param index the start index of the value to be read in {@link #b b}.
734          * @return the read value.
735          */
736         private int readInt(final int index) {
737             byte[] b = this.b;
738             return ((b[index] & 0xFF) << 24) | ((b[index + 1] & 0xFF) << 16)
739                     | ((b[index + 2] & 0xFF) << 8) | (b[index + 3] & 0xFF);
740         }

742         /**
743          * Reads an UTF8 string constant pool item in {@link #b b}. <i>This method
744          * is intended for Attribute sub classes, and is normally not needed
745          * by class generators or adapters.</i>
746          *
747          * @param index the start index of an unsigned short value in {@link #b b},
748          *              whose value is the index of an UTF8 constant pool item.
749          * @param buf   buffer to be used to read the item. This buffer must be
750          *              sufficiently large. It is not automatically resized.
751          * @return the String corresponding to the specified UTF8 item.
752          */
753         private String readUTF8(int index, final char[] buf) {
754             int item = readUnsignedShort(index);
755             String s = strings[item];
756             if (s != null) {
757                 return s;
758             }
759             index = items[item];
760             return strings[item] = readUTF(index + 2, readUnsignedShort(index), buf);
761         }

763         /**
764          * Reads UTF8 string in {@link #b b}.
765          *
766          * @param index  start offset of the UTF8 string to be read.
767          * @param utfLen length of the UTF8 string to be read.
768          * @param buf    buffer to be used to read the string. This buffer must be
769          *               sufficiently large. It is not automatically resized.
770          * @return the String corresponding to the specified UTF8 string.
771          */
772         private String readUTF(int index, final int utfLen, final char[] buf) {
773             int endIndex = index + utfLen;
774             byte[] b = this.b;
775             int strLen = 0;
776             int c;
777             int st = 0;
778             char cc = 0;
779             while (index < endIndex) {
780                 c = b[index++];
781                 switch (st) {
782                     case 0:
783                         c = c & 0xFF;
784                         if (c < 0x80) {  // 0xxxxxxx
785                             buf[strLen++] = (char) c;
786                         } else if (c < 0xE0 && c > 0xBF) {  // 110x xxxx 10xx xxxx
787                             cc = (char) (c & 0x1F);
788                             st = 1;
789                         } else {  // 1110 xxxx 10xx xxxx 10xx xxxx
790                             cc = (char) (c & 0x0F);
791                             st = 2;
792                         }
793                         break;
794 
795                     case 1:  // byte 2 of 2-byte char or byte 3 of 3-byte char
796                         buf[strLen++] = (char) ((cc << 6) | (c & 0x3F));
797                         st = 0;
798                         break;
799 
800                     case 2:  // byte 2 of 3-byte char
801                         cc = (char) ((cc << 6) | (c & 0x3F));
802                         st = 1;
803                         break;
804                     default:
805                         break;
806                 }
807             }
808             return new String(buf, 0, strLen);
809         }

1029             char[] buf = methodDescriptor.toCharArray();
1030             int off = 1;

1032             while (true) {
1033                 char car = buf[off++];
1034                 if (car == ')') {
1035                     break;

1037                     while (buf[off++] != ';') {
1038                     }
