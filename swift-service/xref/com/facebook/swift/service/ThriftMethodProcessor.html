49  @ThreadSafe
50  public class ThriftMethodProcessor
51  {
52      private final String name;
53      private final Object service;
54      private final Method method;
55      private final String resultStructName;
56      private final boolean oneway;
57      private final ImmutableList<ThriftFieldMetadata> parameters;
58      private final Map<Short, ThriftCodec<?>> parameterCodecs;
59      private final Map<Short, Short> thriftParameterIdToJavaArgumentListPositionMap;
60      private final ThriftCodec<Object> successCodec;
61      private final Map<Class<?>, ExceptionProcessor> exceptionCodecs;

63      private final ThriftMethodStats stats = new ThriftMethodStats();

65      public ThriftMethodProcessor(
66              Object service,
67              ThriftMethodMetadata methodMetadata,
68              ThriftCodecManager codecManager
69      )
70      {
71          this.service = service;
72  
73          name = methodMetadata.getName();
74          resultStructName = name + "_result";
75  
76          method = methodMetadata.getMethod();
77          oneway = methodMetadata.getOneway();
78  
79          parameters = ImmutableList.copyOf(methodMetadata.getParameters());
80  
81          ImmutableMap.Builder<Short, ThriftCodec<?>> builder = ImmutableMap.builder();
82          for (ThriftFieldMetadata fieldMetadata : methodMetadata.getParameters()) {
83              builder.put(fieldMetadata.getId(), codecManager.getCodec(fieldMetadata.getType()));
84          }
85          parameterCodecs = builder.build();
86  
87          // Build a mapping from thrift parameter ID to a position in the formal argument list
88          ImmutableMap.Builder<Short, Short> parameterOrderingBuilder = ImmutableMap.builder();
89          short javaArgumentPosition = 0;
90          for (ThriftFieldMetadata fieldMetadata : methodMetadata.getParameters()) {
91              parameterOrderingBuilder.put(fieldMetadata.getId(), javaArgumentPosition++);
92          }
93          thriftParameterIdToJavaArgumentListPositionMap = parameterOrderingBuilder.build();
94  
95          ImmutableMap.Builder<Class<?>, ExceptionProcessor> exceptions = ImmutableMap.builder();
96          for (Map.Entry<Short, ThriftType> entry : methodMetadata.getExceptions().entrySet()) {
97              Class<?> type = TypeToken.of(entry.getValue().getJavaType()).getRawType();
98              ExceptionProcessor processor = new ExceptionProcessor(entry.getKey(), codecManager.getCodec(entry.getValue()));
99              exceptions.put(type, processor);
100         }
101         exceptionCodecs = exceptions.build();
102 
103         successCodec = (ThriftCodec<Object>) codecManager.getCodec(methodMetadata.getReturnType());
104     }

123     public void process(TProtocol in, TProtocol out, int sequenceId)
124             throws Exception
125     {
126         long start = System.nanoTime();
127 
128         // read args
129         Object[] args = readArguments(in);
130 
131         // invoke method
132         Object result;
133         try {
134             result = invokeMethod(args);
135 
136             if (!oneway) {
137                 // write success reply
138                 writeResponse(out,
139                               sequenceId,
140                               TMessageType.REPLY,
141                               "success",
142                               (short) 0,
143                               successCodec,
144                               result);
145             }
146 
147             stats.addSuccessTime(nanosSince(start));
148         }
149         catch (Exception e) {
150             if (!oneway) {
151                 ExceptionProcessor exceptionCodec = exceptionCodecs.get(e.getClass());
152                 if (exceptionCodec != null) {
153                     // write expected exception response
154                     writeResponse(out,
155                                   sequenceId,
156                                   TMessageType.REPLY,
157                                   "exception",
158                                   exceptionCodec.getId(),
159                                   exceptionCodec.getCodec(),
160                                   e);
161                     stats.addErrorTime(nanosSince(start));
162                 } else {
163                     // unexpected exception
164                     TApplicationException applicationException =
165                       new TApplicationException(INTERNAL_ERROR,
166                                                 "Internal error processing " + method.getName());
167                     applicationException.initCause(e);
168 
169                     // Application exceptions are sent to client, and the connection can be reused
170                     out.writeMessageBegin(new TMessage(name, TMessageType.EXCEPTION, sequenceId));
171                     applicationException.write(out);
172                     out.writeMessageEnd();
173                     out.getTransport().flush();
174 
175                     stats.addErrorTime(nanosSince(start));
176                 }
177             } else {
178                 stats.addErrorTime(nanosSince(start));
179             }
180         }
181     }

183     private Object invokeMethod(Object[] args)
184             throws Exception
185     {
186         long start = System.nanoTime();
187         try {
188             Object response = method.invoke(service, args);
189             stats.addInvokeTime(nanosSince(start));
190             if (response instanceof Future)
191             {
192                 // Server-side async isn't implemented yet, so if the server method returns
193                 // a future, we have to wait for it.
194                 return ((Future<?>)response).get();
195             }
196             return response;
197         }
198         catch (Throwable e) {
199             // strip off the InvocationTargetException wrapper if present
200             if (e instanceof InvocationTargetException) {
201                 InvocationTargetException invocationTargetException = (InvocationTargetException) e;
202                 if (invocationTargetException.getTargetException() != null) {
203                     e = invocationTargetException.getTargetException();
204                 }
205             }
206 
207             // rethrow Exceptions or Errors
208             Throwables.propagateIfPossible(e, Exception.class);
209 
210             // Wrap random extensions of Throwable in a runtime exception
211             throw Throwables.propagate(e);
212         }
213     }

215     private Object[] readArguments(TProtocol in)
216             throws Exception
217     {
218         long start = System.nanoTime();
219 
220         try {
221             int numArgs = method.getParameterTypes().length;
222             Object[] args = new Object[numArgs];
223             TProtocolReader reader = new TProtocolReader(in);
224 
225             // Map incoming arguments from the ID passed in on the wire to the position in the
226             // java argument list we expect to see a parameter with that ID.
227             reader.readStructBegin();
228             while (reader.nextField()) {
229                 short fieldId = reader.getFieldId();
230 
231                 ThriftCodec<?> codec = parameterCodecs.get(fieldId);
232                 if (codec == null) {
233                     // unknown field
234                     reader.skipFieldData();
235                 }
236                 else {
237                     // Map the incoming arguments to an array of arguments ordered as the java
238                     // code for the handler method expects to see them
239                     args[thriftParameterIdToJavaArgumentListPositionMap.get(fieldId)] = reader.readField(codec);
240                 }
241             }
242             reader.readStructEnd();
243 
244             // Walk through our list of expected parameters and if no incoming parameters were
245             // mapped to a particular expected parameter, fill the expected parameter slow with
246             // the default for the parameter type.
247             int argumentPosition = 0;
248             for (ThriftFieldMetadata argument : parameters) {
249                 if (args[argumentPosition] == null) {
250                     Type argumentType = argument.getType().getJavaType();
251 
252                     if (argumentType instanceof Class) {
253                         Class<?> argumentClass = (Class<?>) argumentType;
254                         argumentClass = Primitives.unwrap(argumentClass);
255                         args[argumentPosition] = Defaults.defaultValue(argumentClass);
256                     }
257                 }
258                 argumentPosition++;
259             }
260 
261             stats.addReadTime(nanosSince(start));
262             return args;
263         }
264         catch (TProtocolException e) {
265             // TProtocolException is the only recoverable exception
266             // Other exceptions may have left the input stream in corrupted state so we must
267             // tear down the socket.
268             throw new TApplicationException(TApplicationException.PROTOCOL_ERROR, e.getMessage());
269         }
270     }

272     private <T> void writeResponse(TProtocol out,
273                                    int sequenceId,
274                                    byte responseType,
275                                    String responseFieldName,
276                                    short responseFieldId,
277                                    ThriftCodec<T> responseCodec,
278                                    T result) throws Exception {
279         long start = System.nanoTime();
280 
281         out.writeMessageBegin(new TMessage(name, responseType, sequenceId));
282 
283         TProtocolWriter writer = new TProtocolWriter(out);
284         writer.writeStructBegin(resultStructName);
285         writer.writeField(responseFieldName, (short) responseFieldId, responseCodec, result);
286         writer.writeStructEnd();
287 
288         out.writeMessageEnd();
289         out.getTransport().flush();
290 
291         stats.addWriteTime(nanosSince(start));
292     }
