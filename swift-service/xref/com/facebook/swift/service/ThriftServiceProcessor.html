44  /**
45   * Example TProcessor that wraps a Thrift service.  This should only be considered an example, and
46   * is not production ready.  For example, this class makes assumptions about the thrift id of
47   * method parameters, and does not support Thrift exceptions properly.
48   */
49  @ThreadSafe
50  public class ThriftServiceProcessor implements TProcessor
51  {
52      private final Map<String, ThriftMethodProcessor> methods;
53      private final Multimap<String, ThriftMethodStats> serviceStats;
54  
55      /**
56       * @param services the services to expose; services must be thread safe
57       */
58      public ThriftServiceProcessor(ThriftCodecManager codecManager, Object... services)
59      {
60          this(codecManager, ImmutableList.copyOf(services));
61      }
62  
63      public ThriftServiceProcessor(ThriftCodecManager codecManager, List<Object> services)
64      {
65          Preconditions.checkNotNull(codecManager, "codecManager is null");
66          Preconditions.checkNotNull(services, "service is null");
67          Preconditions.checkArgument(!services.isEmpty(), "services is empty");
68  
69          // NOTE: ImmutableMap enforces that we don't have duplicate method names
70          ImmutableMap.Builder<String, ThriftMethodProcessor> processorBuilder = ImmutableMap.builder();
71          ImmutableMultimap.Builder<String, ThriftMethodStats> statsBuilder = ImmutableMultimap.builder();
72          for (Object service : services) {
73              ThriftServiceMetadata serviceMetadata = new ThriftServiceMetadata(service.getClass(), codecManager.getCatalog());
74              for (ThriftMethodMetadata methodMetadata : serviceMetadata.getMethods().values()) {
75                  ThriftMethodProcessor methodProcessor = new ThriftMethodProcessor(service, methodMetadata, codecManager);
76                  processorBuilder.put(methodMetadata.getName(), methodProcessor);
77                  statsBuilder.put(serviceMetadata.getName(), methodProcessor.getStats());
78              }
79          }
80          methods = processorBuilder.build();
81          serviceStats = statsBuilder.build();
82      }
83  
84      public Map<String, ThriftMethodProcessor> getMethods()
85      {
86          return methods;
87      }
88  
89      public Multimap<String, ThriftMethodStats> getServiceStats()
90      {
91          return serviceStats;
92      }
93  
94      @Override
95      @SuppressWarnings("PMD.EmptyCatchBlock")
96      public boolean process(TProtocol in, TProtocol out)
97              throws TException
98      {
99          TMessage message = in.readMessageBegin();
100         String methodName = message.name;
101         int sequenceId = message.seqid;
102 
103         try {
104             // lookup method
105             ThriftMethodProcessor method = methods.get(methodName);
106             if (method == null) {
107                 TProtocolUtil.skip(in, TType.STRUCT);
108                 throw new TApplicationException(UNKNOWN_METHOD, "Invalid method name: '" + methodName + "'");
109             }
110 
111             switch (message.type) {
112                 case TMessageType.CALL:
113                 case TMessageType.ONEWAY:
114                     // Ideally we'd check the message type here to make the presence/absence of
115                     // the "oneway" keyword annotating the method matches the message type.
116                     // Unfortunately most clients send both one-way and two-way messages as CALL
117                     // message type instead of using ONEWAY message type, and servers ignore the
118                     // difference.
119                     break;
120 
121                 default:
122                     throw new TApplicationException(INVALID_MESSAGE_TYPE,
123                                                     "Received invalid message type " + message.type + " from client");
124             }
125 
126             // invoke method
127             method.process(in, out, sequenceId);
128 
129             return true;
130         }
131         catch (Exception e) {
132             // Other exceptions are not recoverable. The input or output streams may be corrupt.
133             Throwables.propagateIfInstanceOf(e, TException.class);
134             throw new TException(e);
135         }
136         finally {
137             try {
138                 in.readMessageEnd();
139             }
140             catch (TException ignore) {
141             }
142         }
143     }
144 }
